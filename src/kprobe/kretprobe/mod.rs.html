166impl<F: KprobeAuxiliaryOps + 'static, L: RawMutex + 'static> From<KretprobeBuilder<L>>
167    for KprobeBuilder<F>
168{
169    fn from(mut value: KretprobeBuilder<L>) -> Self {
170        let kretprobe_data = KretprobeData::<L, F> {
171            free_instances: {
172                let mut instances = Vec::with_capacity(value.maxactive as usize);
173                for _ in 0..value.maxactive {
174                    let data = value.data.pop();
175                    instances.push(KretprobeInstance {
176                        ret_addr: 0,
177                        user_data: data,
178                        frame: 0,
179                        kretprobe: Weak::<Kretprobe<L, F>>::new(),
180                    });
181                }
182                Mutex::new(instances)
183            },
184            kretprobe: Mutex::new(Weak::new()),
185        };
186
187        Self {
188            symbol: value.symbol.clone(),
189            symbol_addr: value.symbol_addr,
190            offset: 0,
191            pre_handler: Some(ProbeHandler::new(pre_handler_kretprobe::<L, F>)),
192            post_handler: None,
193            fault_handler: None,
194            event_callbacks: BTreeMap::new(),
195            probe_point: None,
196            enable: true,
197            data: Some(Box::new(kretprobe_data)),
198            _marker: core::marker::PhantomData,
199        }
200    }
201}

203fn pre_handler_kretprobe<L: RawMutex + 'static, F: KprobeAuxiliaryOps + 'static>(
204    data: &(dyn ProbeData),
205    pt_regs: &mut PtRegs,
206) {
207    let kretprobe_data = data.as_any().downcast_ref::<KretprobeData<L, F>>().unwrap();
208    let free_instance = kretprobe_data.free_instances.lock().pop();
209    if let Some(mut instance) = free_instance {
210        arch_rethook_prepare::<L, F>(&mut instance, pt_regs);
211        let user_data = instance.user_data.as_deref().unwrap_or(&());
212
213        let kretprobe = kretprobe_data.kretprobe.lock();
214        let kretprobe = kretprobe.upgrade().unwrap();
215        if let Some(entry_handler) = kretprobe.entry_handler {
216            entry_handler.call(user_data, pt_regs);
217        }
218        instance.kretprobe = Arc::downgrade(&(kretprobe as Arc<dyn Any + Send + Sync>));
219
220        // insert the instance into the task
221        F::insert_kretprobe_instance_to_task(instance);
222    } else {
223        log::warn!("No free KretprobeInstance available in pre_handler_kretprobe");
224        let kretprobe = kretprobe_data.kretprobe.lock();
225        let kretprobe = kretprobe.upgrade();
226        let Some(kretprobe) = kretprobe else {
227            panic!("Kretprobe is not available in pre_handler_kretprobe");
228        };
229        kretprobe
230            .nmissed
231            .fetch_add(1, core::sync::atomic::Ordering::Relaxed);
232    }
233}

235pub(crate) fn rethook_trampoline_handler<L: RawMutex + 'static, F: KprobeAuxiliaryOps + 'static>(
236    pt_regs: &mut PtRegs,
237    frame: usize,
238) -> usize {
239    let kretprobe_instance = F::pop_kretprobe_instance_from_task();
240    let correct_ret_addr = kretprobe_instance.ret_addr;
241    assert_eq!(kretprobe_instance.frame, frame);
242    let user_data = kretprobe_instance.user_data.as_deref().unwrap_or(&());
243
244    let kretprobe = kretprobe_instance.kretprobe.upgrade().unwrap();
245    let kretprobe = kretprobe
246        .as_ref()
247        .downcast_ref::<Kretprobe<L, F>>()
248        .unwrap();
249
250    // call the return handler if it exists
251    if let Some(ret_handler) = kretprobe.ret_handler {
252        ret_handler.call(user_data, pt_regs);
253    }
254
255    // call the event callbacks if they exist
256    for callback in kretprobe.event_callbacks.lock().values() {
257        callback.call(pt_regs);
258    }
259
260    arch_rethook_fixup_return(pt_regs, correct_ret_addr);
261    correct_ret_addr
