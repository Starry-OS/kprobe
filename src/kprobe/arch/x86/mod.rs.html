25/// The probe point for x86_64 architecture.
26#[derive(Debug)]
27pub struct X86KprobePoint<F: KprobeAuxiliaryOps> {
28    addr: usize,
29    old_instruction_ptr: usize,
30    old_instruction_len: usize,
31    _marker: core::marker::PhantomData<F>,
32}
33
34impl<F: KprobeAuxiliaryOps> Drop for X86KprobePoint<F> {
35    fn drop(&mut self) {
36        let address = self.addr;
37        unsafe {
38            F::set_writeable_for_address(address, self.old_instruction_len, true);
39            core::ptr::copy(
40                self.old_instruction_ptr as *const u8,
41                address as *mut u8,
42                self.old_instruction_len,
43            );
44            F::set_writeable_for_address(address, self.old_instruction_len, false);
45            core::arch::x86_64::_mm_mfence();
46        }
47        let decoder = yaxpeax_x86::amd64::InstDecoder::default();
48        let buf = unsafe {
49            core::slice::from_raw_parts(
50                self.old_instruction_ptr as *const u8,
51                self.old_instruction_len,
52            )
53        };
54        let inst = decoder.decode_slice(buf).unwrap();
55
56        F::dealloc_executable_memory(
57            self.old_instruction_ptr as *mut u8,
58            Layout::from_size_align(MAX_INSTRUCTION_SIZE + 1, 16).unwrap(),
59        );
60        log::trace!(
61            "Kprobe::uninstall: address: {:#x}, old_instruction: {:?}",
62            address,
63            inst.to_string()
64        );
65    }
66}

91impl<F: KprobeAuxiliaryOps> KprobeBuilder<F> {
92    pub(crate) fn install<L: RawMutex + 'static>(self) -> (Kprobe<L, F>, Arc<X86KprobePoint<F>>) {
93        let probe_point = match &self.probe_point {
94            Some(point) => point.clone(),
95            None => self.replace_inst(),
96        };
97        let kprobe = Kprobe {
98            basic: KprobeBasic::from(self),
99            point: probe_point.clone(),
100        };
101        (kprobe, probe_point)
102    }
103
104    /// Replace the instruction at the specified address with a breakpoint instruction.
105    fn replace_inst(&self) -> Arc<X86KprobePoint<F>> {
106        let address = self.symbol_addr + self.offset;
107        let inst_tmp = F::alloc_executable_memory(
108            Layout::from_size_align(MAX_INSTRUCTION_SIZE + 1, 16).unwrap(),
109        );
110        unsafe {
111            core::ptr::copy(address as *const u8, inst_tmp, MAX_INSTRUCTION_SIZE);
112        }
113
114        let decoder = yaxpeax_x86::amd64::InstDecoder::default();
115        let buf = unsafe { core::slice::from_raw_parts(inst_tmp, MAX_INSTRUCTION_SIZE) };
116        let inst = decoder.decode_slice(buf).unwrap();
117        let len = inst.len().to_const();
118        log::trace!("inst: {:?}, len: {:?}", inst.to_string(), len);
119
120        let point = Arc::new(X86KprobePoint {
121            addr: address,
122            old_instruction_ptr: inst_tmp as usize,
123            old_instruction_len: len as usize,
124            _marker: core::marker::PhantomData,
125        });
126
127        unsafe {
128            F::set_writeable_for_address(address, len as usize, true);
129            core::ptr::write_volatile(address as *mut u8, EBREAK_INST);
130            F::set_writeable_for_address(address, len as usize, false);
131            core::arch::x86_64::_mm_mfence();
132        }
133
134        log::trace!(
135            "Kprobe::install: address: {:#x}, func_name: {:?}",
136            address,
137            self.symbol
138        );
139        point
140    }
141}

211impl PtRegs {
212    pub(crate) fn break_address(&self) -> usize {
213        self.rip - 1 // The breakpoint instruction is at the address of rip - 1
214    }
215
216    pub(crate) fn debug_address(&self) -> usize {
217        self.rip // The debug address is the current instruction pointer
218    }
219
220    pub(crate) fn update_pc(&mut self, pc: usize) {
221        self.rip = pc as _;
222    }
223
224    pub(crate) fn set_single_step(&mut self, enable: bool) {
225        if enable {
226            self.rflags |= 0x100;
227        } else {
228            self.rflags &= !0x100;
229        }
230    }
231
232    pub(crate) fn sp(&self) -> usize {
233        self.rsp
234    }
235
236    /// Get the return value from the rax register.
237    pub fn ret_value(&self) -> usize {
238        self.rax
239    }

242#[unsafe(naked)]
243pub(crate) unsafe extern "C" fn arch_rethook_trampoline<
244    L: RawMutex + 'static,
245    F: KprobeAuxiliaryOps + 'static,
246>() {
247    core::arch::naked_asm!(
248        // Push a fake return address to tell the unwinder it's a kretprobe.
249        "pushq 0",
250        "pushq {kernel_data_segment}", // fake ss
251        // Save the 'sp - 16', this will be fixed later.
252        "pushq %rsp",
253        "pushfq", // rflags
254
255        // SAVE_REGS_STRING
256        "subq $24, %rsp", // skip cs, ip, orig_ax
257        "pushq %rdi",
258        "pushq %rsi",
259        "pushq %rdx",
260        "pushq %rcx",
261        "pushq %rax",
262        "pushq %r8",
263        "pushq %r9",
264        "pushq %r10",
265        "pushq %r11",
266        "pushq %rbx",
267        "pushq %rbp",
268        "pushq %r12",
269        "pushq %r13",
270        "pushq %r14",
271        "pushq %r15",
272        // ENCODE_FRAME_POINTER
273        "lea 1(%rsp), %rbp",
274        "movq %rsp, %rdi",
275        "call {rethook_trampoline_callback}",
276        // RESTORE_REGS_STRING
277        "popq %r15",
278        "popq %r14",
279        "popq %r13",
280        "popq %r12",
281        "popq %rbp",
282        "popq %rbx",
283        "popq %r11",
284        "popq %r10",
285        "popq %r9",
286        "popq %r8",
287        "popq %rax",
288        "popq %rcx",
289        "popq %rdx",
290        "popq %rsi",
291        "popq %rdi",
292        // Skip orig_ax, ip, cs
293        "addq $24, %rsp",
294
295        // In the callback function, 'regs->flags' is copied to 'regs->ss'.
296        "addq $16, %rsp",
297
298        "popfq",
299        "ret",
300        kernel_data_segment = const 0x10, // Kernel data segment
301        // arch_rethook_trampoline = sym arch_rethook_trampoline::<L,F>,
302        rethook_trampoline_callback = sym arch_rethook_trampoline_callback::<L, F>,
303        options(att_syntax)
304    )
305}

307pub(crate) fn arch_rethook_trampoline_callback<
308    L: RawMutex + 'static,
309    F: KprobeAuxiliaryOps + 'static,
310>(
311    pt_regs: &mut PtRegs,
312) -> usize {
313    pt_regs.rip = arch_rethook_trampoline::<L, F> as usize; // Set return address to trampoline
314    pt_regs.orig_rax = usize::MAX;
315    pt_regs.rsp += 16; // Adjust rsp to remove the fake return address
316
317    let pt_regs_pointer = unsafe { (pt_regs as *mut PtRegs).add(1) as *mut usize };
318
319    let correct_ret_addr = rethook_trampoline_handler::<L, F>(pt_regs, pt_regs_pointer as _);
320    pt_regs.ss = pt_regs.rflags; // Copy eflags to ss
321    correct_ret_addr
322}

324/// Fix up the return address in the pt_regs after a rethook.
325pub(crate) fn arch_rethook_fixup_return(pt_regs: &mut PtRegs, correct_ret_addr: usize) {
326    let pt_regs_pointer = unsafe { (pt_regs as *mut PtRegs).add(1) as *mut usize };
327    unsafe {
328        // Replace fake return address with real one.
329        *pt_regs_pointer = correct_ret_addr;
330    }
331}

333/// Prepare the kretprobe instance for the rethook.
334pub(crate) fn arch_rethook_prepare<L: RawMutex + 'static, F: KprobeAuxiliaryOps + 'static>(
335    kretprobe_instance: &mut KretprobeInstance,
336    pt_regs: &mut PtRegs,
337) {
338    let sp = pt_regs.sp();
339    let stack = unsafe { &mut *(sp as *mut usize) };
340    // Prepare the kretprobe instance for the rethook
341    // Get the return address from the stack
342    kretprobe_instance.ret_addr = *stack;
343    kretprobe_instance.frame = sp;
344    *stack = arch_rethook_trampoline::<L, F> as _; // Set the return address to the trampoline
345}
