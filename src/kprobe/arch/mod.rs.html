34/// The operations available for kprobes.
35pub trait KprobeOps: Send {
36    /// The address of the instruction that program should return to
37    fn return_address(&self) -> usize;
38
39    /// The address of the instruction that saved the original instruction
40    ///
41    /// Usually, the original instruction at the probe point is saved in an array.
42    /// Depending on the architecture, necessary instructions may be filled in after
43    /// the saved instruction. For example, x86 architecture supports single-step execution,
44    /// while other architectures usually do not. Therefore, we use the break exception to
45    /// simulate it, so a breakpoint instruction will be filled in.
46    fn single_step_address(&self) -> usize;
47
48    /// The address of the instruction that caused the single step exception
49    fn debug_address(&self) -> usize;
50
51    /// The address of the instruction that caused the break exception
52    ///
53    /// It is usually equal to the address of the instruction that used to set the probe point.
54    fn break_address(&self) -> usize;
55}

77/// The type of the probe handler function.
78pub type ProbeHandlerFunc = fn(&(dyn ProbeData), &mut PtRegs);

80#[derive(Clone, Copy, Debug)]
81pub(crate) struct ProbeHandler {
82    pub(crate) func: ProbeHandlerFunc,
83}
84
85impl ProbeHandler {
86    pub fn new(func: ProbeHandlerFunc) -> Self {
87        ProbeHandler { func }
88    }
89
90    pub fn call(&self, data: &(dyn ProbeData), pt_regs: &mut PtRegs) {
91        (self.func)(data, pt_regs);
92    }
93}

101/// The builder for creating a kprobe.

116impl<F: KprobeAuxiliaryOps> KprobeBuilder<F> {
117    /// Create a new kprobe builder.
118    pub fn new(symbol: Option<String>, symbol_addr: usize, offset: usize, enable: bool) -> Self {
119        KprobeBuilder {
120            symbol,
121            symbol_addr,
122            offset,
123            pre_handler: None,
124            post_handler: None,
125            event_callbacks: BTreeMap::new(),
126            fault_handler: None,
127            probe_point: None,
128            enable,
129            data: None,
130            _marker: core::marker::PhantomData,
131        }
132    }
133
134    /// Build the kprobe with a specific user data.
135    pub fn with_data<T: ProbeData>(mut self, data: T) -> Self {
136        self.data = Some(Box::new(data));
137        self
138    }
139
140    /// Build the kprobe with a pre handler function.
141    pub fn with_pre_handler(mut self, func: ProbeHandlerFunc) -> Self {
142        self.pre_handler = Some(ProbeHandler::new(func));
143        self
144    }
145
146    /// Build the kprobe with a post handler function.
147    pub fn with_post_handler(mut self, func: ProbeHandlerFunc) -> Self {
148        self.post_handler = Some(ProbeHandler::new(func));
149        self
150    }
151
152    /// Build the kprobe with a pre handler function.
153    pub fn with_fault_handler(mut self, func: ProbeHandlerFunc) -> Self {
154        self.fault_handler = Some(ProbeHandler::new(func));
155        self
156    }
157
158    pub(crate) fn with_probe_point(mut self, point: Arc<KprobePoint<F>>) -> Self {
159        self.probe_point = Some(point);
160        self
161    }
162
163    /// Build the kprobe with an event callback function.
164    pub fn with_event_callback(
165        mut self,
166        callback_id: u32,
167        event_callback: Box<dyn CallBackFunc>,
168    ) -> Self {
169        self.event_callbacks.insert(callback_id, event_callback);
170        self
171    }
172
173    /// Get the address of the instruction that should be probed.
174    pub fn probe_addr(&self) -> usize {
175        self.symbol_addr + self.offset
176    }

179/// The basic information of a kprobe.
180pub struct KprobeBasic<L: RawMutex + 'static> {
181    symbol: Option<String>,
182    symbol_addr: usize,
183    offset: usize,
184    pre_handler: Option<ProbeHandler>,
185    post_handler: Option<ProbeHandler>,
186    fault_handler: Option<ProbeHandler>,
187    event_callbacks: Mutex<L, BTreeMap<u32, Box<dyn CallBackFunc>>>,
188    enable: AtomicBool,
189    data: Box<dyn ProbeData>,
190}

192impl<L: RawMutex + 'static> Debug for KprobeBasic<L> {
193    fn fmt(&self, f: &mut core::fmt::Formatter<'_>) -> core::fmt::Result {
194        f.debug_struct("Kprobe")
195            .field("symbol", &self.symbol)
196            .field("symbol_addr", &self.symbol_addr)
197            .field("offset", &self.offset)

202impl<L: RawMutex + 'static> KprobeBasic<L> {
203    /// Call the pre handler function.
204    pub fn call_pre_handler(&self, pt_regs: &mut PtRegs) {
205        if let Some(ref handler) = self.pre_handler {
206            handler.call(self.data.as_ref(), pt_regs);
207        }
208    }
209
210    /// Call the post handler function.
211    pub fn call_post_handler(&self, pt_regs: &mut PtRegs) {
212        if let Some(ref handler) = self.post_handler {
213            handler.call(self.data.as_ref(), pt_regs);
214        }
215    }
216
217    /// Call the fault handler function.
218    pub fn call_fault_handler(&self, pt_regs: &mut PtRegs) {
219        if let Some(ref handler) = self.fault_handler {
220            handler.call(self.data.as_ref(), pt_regs);
221        }
222    }
223
224    /// Call the event callback function.
225    pub fn call_event_callback(&self, pt_regs: &mut PtRegs) {
226        let event_callbacks = self.event_callbacks.lock();
227        for callback in event_callbacks.values() {
228            callback.call(pt_regs);
229        }
230    }
231
232    /// Register the event callback function.
233    pub fn register_event_callback(&self, callback_id: u32, callback: Box<dyn CallBackFunc>) {
234        self.event_callbacks.lock().insert(callback_id, callback);
235    }
236
237    /// Unregister the event callback function.
238    pub fn unregister_event_callback(&self, callback_id: u32) {
239        self.event_callbacks.lock().remove(&callback_id);
240    }
241
242    /// Disable the probe point.
243    pub fn disable(&self) {
244        self.enable.store(false, Ordering::Relaxed);
245    }
246
247    /// Enable the probe point.
248    pub fn enable(&self) {
249        self.enable.store(true, Ordering::Relaxed);
250    }
251
252    /// Check if the probe point is enabled.
253    pub fn is_enabled(&self) -> bool {
254        self.enable.load(Ordering::Relaxed)
255    }
256
257    /// Get the function name of the probe point.
258    pub fn symbol(&self) -> Option<&str> {
259        self.symbol.as_deref()
260    }
261
262    pub(crate) fn get_data(&self) -> &(dyn ProbeData) {
263        self.data.as_ref()
264    }
265}
