7/// A manager for kprobes.
8#[derive(Debug)]
9pub struct KprobeManager<L: RawMutex + 'static, F: KprobeAuxiliaryOps> {
10    break_list: BTreeMap<usize, Vec<Probe<L, F>>>,
11    debug_list: BTreeMap<usize, Vec<Probe<L, F>>>,
12}
13
14impl<L: RawMutex + 'static, F: KprobeAuxiliaryOps> Default for KprobeManager<L, F> {
15    fn default() -> Self {
16        Self::new()
17    }
18}

20impl<L: RawMutex + 'static, F: KprobeAuxiliaryOps> KprobeManager<L, F> {
21    /// Create a new kprobe manager.
22    pub const fn new() -> Self {
23        KprobeManager {
24            break_list: BTreeMap::new(),
25            debug_list: BTreeMap::new(),
26        }
27    }
28    /// Insert a kprobe into the manager.
29    pub fn insert_probe(&mut self, probe: Probe<L, F>) {
30        let probe_point = probe.probe_point().clone();
31        self.insert_break_point(probe_point.break_address(), probe.clone());
32        self.insert_debug_point(probe_point.debug_address(), probe);
33    }
34
35    /// Insert a kprobe into the break_list.
36    ///
37    /// # Parameters
38    /// - `address`: The address of the kprobe, obtained from `KprobePoint::break_address()` or `KprobeBuilder::probe_addr()`.
39    /// - `kprobe`: The instance of the kprobe.
40    fn insert_break_point(&mut self, address: usize, probe: Probe<L, F>) {
41        let list = self.break_list.entry(address).or_default();
42        list.push(probe);
43    }
44
45    /// Insert a kprobe into the debug_list.
46    ///
47    /// # Parameters
48    /// - `address`: The address of the kprobe, obtained from `KprobePoint::debug_address()`.
49    /// - `kprobe`: The instance of the kprobe.
50    ///
51    fn insert_debug_point(&mut self, address: usize, probe: Probe<L, F>) {
52        let list = self.debug_list.entry(address).or_default();
53        list.push(probe);
54    }
55
56    /// Get the list of kprobes registered at a breakpoint address.
57    pub fn get_break_list(&self, address: usize) -> Option<&Vec<Probe<L, F>>> {
58        self.break_list.get(&address)
59    }
60
61    /// Get the list of kprobes registered at a debug address.
62    pub fn get_debug_list(&self, address: usize) -> Option<&Vec<Probe<L, F>>> {
63        self.debug_list.get(&address)
64    }
65
66    /// Get the number of kprobes registered at a breakpoint address.
67    pub fn kprobe_num(&self, address: usize) -> usize {
68        self.break_list_len(address)
69    }
70
71    #[inline]
72    fn break_list_len(&self, address: usize) -> usize {
73        self.break_list
74            .get(&address)
75            .map(|list| list.len())
76            .unwrap_or(0)
77    }
78    #[inline]
79    fn debug_list_len(&self, address: usize) -> usize {
80        self.debug_list
81            .get(&address)
82            .map(|list| list.len())
83            .unwrap_or(0)
84    }
85
86    /// Remove a kprobe from the manager.
87    pub fn remove_kprobe(&mut self, probe: &Probe<L, F>) {
88        let probe_point = probe.probe_point().clone();
89        self.remove_one_break(probe_point.break_address(), probe);
90        self.remove_one_debug(probe_point.debug_address(), probe);
91    }
92
93    /// Remove a kprobe from the break_list.
94    fn remove_one_break(&mut self, address: usize, probe: &Probe<L, F>) {
95        if let Some(list) = self.break_list.get_mut(&address) {
96            list.retain(|x| match x {
97                Probe::Kprobe(kprobe) => {
98                    match probe {
99                        Probe::Kprobe(kprobe2) => !Arc::ptr_eq(kprobe, kprobe2),
100                        Probe::Kretprobe(_) => true, // Kretprobe should not match Kprobe
101                    }
102                }
103                Probe::Kretprobe(kretprobe) => {
104                    match probe {
105                        Probe::Kprobe(_) => true, // Kprobe should not match Kretprobe
106                        Probe::Kretprobe(kretprobe2) => !Arc::ptr_eq(kretprobe, kretprobe2),
107                    }
108                }
109            });
110        }
111        if self.break_list_len(address) == 0 {
112            self.break_list.remove(&address);
113        }
114    }
115
116    /// Remove a kprobe from the debug_list.
117    fn remove_one_debug(&mut self, address: usize, probe: &Probe<L, F>) {
118        if let Some(list) = self.debug_list.get_mut(&address) {
119            list.retain(|x| match x {
120                Probe::Kprobe(kprobe) => {
121                    match &probe {
122                        Probe::Kprobe(kprobe2) => !Arc::ptr_eq(kprobe, kprobe2),
123                        Probe::Kretprobe(_) => true, // Kretprobe should not match Kprobe
124                    }
125                }
126                Probe::Kretprobe(kretprobe) => {
127                    match &probe {
128                        Probe::Kprobe(_) => true, // Kprobe should not match Kretprobe
129                        Probe::Kretprobe(kretprobe2) => !Arc::ptr_eq(kretprobe, kretprobe2),
130                    }
131                }
132            });
133        }
134        if self.debug_list_len(address) == 0 {
135            self.debug_list.remove(&address);
136        }
137    }
138}
